## install for  linux
***
```
sudo apt update && apt upgrade

sudo apt install clang gcc -y

git clone https://github.com/RedMads/tic_tac_toe.git

cd tic_tac_toe

sudo mv conio.h /usr/include

g++ main.cpp -o Tic_tac_toe

chmod +x Tic_tac_toe

./Tic_tac_toe
```
